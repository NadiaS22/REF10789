#include <iostream>
using namespace std;

int main() {

    float nota1, nota2, nota3, nota4, media;

    cout << "Indique-me 4 notas.\n";
    cout << "Primeira nota:";
    cin >> nota1;
    cout << "Segunda nota:";
    cin >> nota2;
    cout << "Terceira nota:";
    cin >> nota3;
    cout << "Quarta nota:";
    cin >> nota4;
    
    media = (nota1 + nota2 + nota3 + nota4) / 4;
    
    cout << "As notas indicadas foram " << nota1 << "," << nota2 << "," << nota3 << "," << nota4 << ", cuja média é " << media << endl;

    return 0;
}
